Login

Request

URL

https://payon.id/api/main/login

Method

POST or GET

Parameters

  • username : (Required) Username akun
  • password : (Required) Password akun

Response

{
"success": true,
"results": {
    "otp": "",
    "id": "1234",
    "name": "My Name",
    "username": "payonid",
    "balance": "10000",
    "token": "1234:xM3pN0FdotaTinGbLwhz7v8RfKC4SYgU"
    }
}

Example


?php
    
$parameters = array(
    "username" => "",
    "password" => "",
    );

$ch = curl_init();
curl_setopt($ch, CURLOPT_URL, "https://payon.id/api/main/login");
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
curl_setopt($ch, CURLOPT_POST, true);
curl_setopt($ch, CURLOPT_POSTFIELDS, http_build_query($parameters));
$response = curl_exec($ch);
curl_close($ch);

Rincian Akun

Request

URL

https://payon.id/api/main/account

Method

POST or GET

Parameters

  • auth_username : (Required) Username akun
  • auth_token : (Required) Token yang didapat ketika berhasil login

Response

{
    {
    "success": true,
    "results": {
        "id": 1234,
        "username": "payonid",
        "name": "My Name",
        "email": "mail@domain.com",
        "gender": "Laki-laki",
        "address": "Tangerang",
        "phone": "081213141516",
        "balance": 10000,
        "balance_str": "Rp 10.000",
        "status": "Terverifikasi",
        "type": "Member",
        "data": {
            "phone_verified": true,
            "email_verified": true,
            "otp": "",
            "login_notify": false,
            "order_verify_password": true
        },
        "total_trx": "0",
        "reg_date": "21/01/2019 02:41"
    }
}

Example

?php

$parameters = array(

Pembelian

Request

URL

https://payon.id/api/main/order

Method

POST or GET

Parameters

  • auth_username : (Required) Username akun
  • auth_token : (Required) Token yang didapat ketika berhasil login
  • voucher_id : (Required) ID Voucher lihat di daftar harga (id)
  • phone : (Required) Nomor HP pembeli atau nomor HP yang akan diisi ulang
  • id_plgn : (Optional) ID Pelanggan untuk order token PLN / Tagihan
  • payment : (Required) Static=balance

Response

{
"success": true,
"status": true,
"results": {
    "id": 240,
    "no_hp": "083812345678",
    "id_plgn": "",
    "harga": 7625,
    "harga_str": "Rp 7.625",
    "pembayaran": "Saldo Akun",
    "status": "240",
    "tanggal": "12/02/2019 00:50",
    "produk": {
        "id": "paket_internet",
        "nama": "Paket Internet"
    },
    "provider": {
        "id": 10,
        "nama": "AXIS DATA BRONET"
    },
    "voucher": {
        "id": 71,
        "nominal": "BRONET 300MB 7Hr"
    }
}
}

Example


?php

$parameters = array(
"auth_username" => "payonid",
"auth_token" => "1234:xM3pN0FdotaTinGbLwhz7v8RfKC4SYgU",
"voucher_id" => "71",
"phone" => "083812345678",
"id_plgn" => "",
"payment" => "balance",
);

$ch = curl_init();
curl_setopt($ch, CURLOPT_URL, "https://payon.id/api/main/order");
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
curl_setopt($ch, CURLOPT_POST, true);
curl_setopt($ch, CURLOPT_POSTFIELDS, http_build_query($parameters));
$response = curl_exec($ch);
curl_close($ch);
